git init - команда для того, чтобы сделать каталог репрозиторием
git status - команда для вывода информации о изменениях
git add - команда для добавления файлов в репрозиторий
git commit -m "" - команда для написания сообщения об изменениях
git log - команда для просмотра истории изменений
git branch %имя-ветки% - создать новую ветку
git branch - показать список веток
git switch %имя-ветки% - перейти на ветку (для новых версий)
git checkout %имя-ветки% - перейти на ветку (для старых версий)
git merge %имя-ветки%. - слить указанную ветку в текущую
git branch -d %имя-ветки% - удалить ветку
